56/// ∑ wᵢ(aᵢ - bᵢ)².
57template <class T, class Abi, StorageOrder OW, StorageOrder OA, StorageOrder OB>
58[[gnu::flatten]] T weighted_norm_sq_difference(view<const T, Abi, OW> w, view<const T, Abi, OA> a,
59 view<const T, Abi, OB> b) {
